from collections import deque

def countZero(board):
    cnt = 0
    for row in board:
        cnt += row.count(0)
    return cnt

M, N = map(int, input().split())
board = [list(map(int, input().split())) for _ in range(N)]

q = deque()
dx, dy = [0,-1,0,1], [1,0,-1,0]
ans = 0
flag = False # í† ë§ˆí† ê°€ ëª¨ë‘ ìµì„ ìˆ˜ ìˆëŠ” ìƒí™©ì¸ì§€
for i in range(N):
    for j in range(M):
        if board[i][j] == 1:
            q.append((i, j))
while q:
    if countZero(board) == 0:
        flag = True
        break
    for _ in range(len(q)):
        cur = q.popleft()
        for w in range(4):
            xx, yy = cur[0]+dx[w], cur[1]+dy[w]
            if xx<0 or xx>=N or yy<0 or yy>=M:
                continue
            if board[xx][yy] == 0:
                board[xx][yy] = 1
                q.append((xx, yy))
    ans += 1

print([-1, ans][flag])
