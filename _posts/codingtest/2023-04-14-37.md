from collections import deque

def solution(begin, target, words):
    answer = 0
    q = deque()
    q.append([begin, 0])
    visited = [0 for _ in range(len(words))]
    
    if target not in words:
        return 0
    
    while q:
        word, cnt = q.popleft()
        if word == target:
            answer = cnt
            break
        for i in range(len(words)):
            temp_cnt = 0
            if not visited[i]:
                for j in range(len(word)):
                    if word[j] != words[i][j]:
                        temp_cnt += 1
                if temp_cnt == 1: # í•œ ê¸€ìë§Œ ë‹¤ë¥¸ ê²½ìš°
                    q.append([words[i], cnt+1]) # íì— ë„£ê¸°

    return answer
