### 내 풀이
```py
import sys

dic = {
    '-': [0,0,0,0,0,0,0], # 맨앞 0
    0: [1,1,0,1,1,1,1], # 중간 0
    1: [0,0,0,1,0,0,1],
    2: [1,0,1,1,1,1,0],
    3: [1,0,1,1,0,1,1],
    4: [0,1,1,1,0,0,1],
    5: [1,1,1,0,0,1,1],
    6: [1,1,1,0,1,1,1],
    7: [1,1,0,1,0,0,1],
    8: [1,1,1,1,1,1,1],
    9: [1,1,1,1,0,1,1]
}

def reverse(a): # a: string (ex. '123')
    a = a[::-1] # 거꾸로 뒤집기 # 321
    lst_A = ['-'] * 5 # 맨 앞 0을 구분하기 위해 - 으로!
    for i in range(4, 5-len(a)-1, -1):
        lst_A[i] = int(a[4-i]) # 4,3,2 -> 0(4-4),1(4-3),2(4-2)
    return lst_A # [-,-,1,2,3]

def compareOneDigit(a, b):
    cnt = 0
    for i in range(7):
        if dic[a][i] != dic[b][i]:
            cnt += 1
    return cnt

def process():
    a, b = input().split()
    lst_A = reverse(a)
    lst_B = reverse(b)

    cnt = 0
    for i in range(5):
        cnt += compareOneDigit(lst_A[i], lst_B[i])
    print(cnt)

TC = int(input())
for _ in range(TC):
    process()
```
