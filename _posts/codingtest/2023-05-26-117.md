from collections import deque

def solution(maps):
    N = len(maps[0])
    M = len(maps)
    dist = [[0] * N for _ in range(M)]
    visited = [[0] * N for _ in range(M)]
    
    q = deque()

        for w in range(4):
            xx, yy = x+dx[w], y+dy[w]
            if xx<0 or xx>=N or yy<0 or yy>=M:

    ans = dist[0][0]
