```py
from collections import deque

def solution(n, computers):
    visited = [0] * n
    dist = [0] * n
    
    q = deque()
    q.append(0)
    visited[0] = 1
    while sum(visited) != n:
        if q:
            cur = q.popleft()
            for i in range(n):
                if computers[cur][i] == 1 and not visited[i]:
                    q.append(i)
                    visited[i] = 1
                    dist[i] = dist[cur] + 1
        else:
            q.append(visited.index(0))
            visited[i] = 1
    return dist.count(0)
```
