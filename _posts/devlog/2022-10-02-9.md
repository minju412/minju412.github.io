public enum BoardType {
    HOT, FREE, PROTECTOR, JOB;

    @JsonCreator
    public static BoardType from(String str) {
        String res = "";
        switch (str.toUpperCase()) {
            case "HOT":
                res = "HOT";
                break;
            case "FRE":
                res = "FREE";
                break;
            case "PRO":
                res = "PROTECTOR";
                break;
            case "JOB":
                res = "JOB";
                break;
            default:
                throw new CustomException(NO_MATCHING_ENUM_TYPE);
        }
        return BoardType.valueOf(res.toUpperCase());
    }
}
