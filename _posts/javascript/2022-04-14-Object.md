# for..in vs for..of
```js
const ann = {name: 'ann', age: 20}; 

// for..in (key in obj)
for(key in ann){
  console.log(key); // name age
}

// for..of (value of iterable)
const arr = [1,2,3,4];
for(value of arr){
  console.log(value); // 1 2 3 4
}
```
