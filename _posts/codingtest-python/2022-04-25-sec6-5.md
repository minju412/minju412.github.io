## 풀이 - 1
```py
import sys
sys.stdin = open("./input/in5.txt", "rt")

def DFS(i,sum): # i는 인덱스 번호, sum은 원소의 누적 합
    global max # 전역변수 max 사용
    if sum>c: # c를 넘지 않는 누적합을 구하기 위한 조건
        return
    if i==n: # 부분집합 하나 완성
        if sum>max:
            max=sum 
    else:
        DFS(i+1,sum+lst[i])
        DFS(i+1,sum)
    


if __name__=="__main__":
    c,n=map(int,input().split())
    lst=[0]*n
    for i in range(n):
        lst[i]=int(input())

    max=0
    DFS(0,0)
    print(max)
```
